Quick Start

Basic Usage 
# Create an octree with 0.1m resolution
import pyoctomap
import numpy as np

tree = pyoctomap.OcTree(0.1)

# Add occupied points
tree.updateNode([1.0, 2.0, 3.0], True)
tree.updateNode([1.1, 2.1, 3.1], True)

# Add free space
tree.updateNode([0.5, 0.5, 0.5], False)

# Check occupancy
node = tree.search([1.0, 2.0, 3.0])
if node and tree.isNodeOccupied(node):
    print("Point is occupied!")

# Save to file
tree.write("my_map.bt")
Advanced: Ray Casting with Point Clouds 
# High-performance batch processing with ray casting
points = np.random.uniform(-5, 5, (1000, 3))
origin = np.array([0., 0., 0.], dtype=np.float64)

# Precise batch ray-casting (hit-stopping for free/occupied)
success_count = tree.addPointCloudWithRayCasting(points, origin, discretize=True)

# Fast native batching (full rays, optional discretization)
tree.insertPointCloudFast(points, origin, discretize=False, lazy_eval=True)
tree.updateInnerOccupancy()  # Manual after lazy

# Iterate over occupancy data
for leaf_it in tree.begin_leafs():
    coord = leaf_it.getCoordinate()
    if tree.isNodeOccupied(leaf_it):
        print(f"Occupied leaf at {coord}")

Installation

pip (Recommended)
From Source
Quick Install 
# Install pre-built wheel (recommended)
pip install pyoctomap

# With visualization dependencies
pip install pyoctomap[visualization]

# For development
pip install pyoctomap[dev]

Supported: Python 3.9, 3.10, 3.11, 3.12 | Linux (manylinux2014) | Windows (WSL)

Build from Source 
# Clone with submodules
git clone --recursive https://github.com/Spinkoo/pyoctomap.git
cd pyoctomap

# Build OctoMap C++ library
cd src/octomap
mkdir build && cd build
cmake .. && make && sudo make install

# Build Python package
cd ../../..
python setup.py build_ext --inplace
